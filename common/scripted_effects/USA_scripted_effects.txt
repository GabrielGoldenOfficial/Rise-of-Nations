USA_campaign_timeout_effect = {
	hidden_effect = {
		calcelccollage = yes
	}
	if = {
		limit = {
			OR = {
			check_variable = { republican_vote > 270 }
			check_variable = { republican_vote > democratic_vote }				
			}
		}
		custom_effect_tooltip = gopprojectedwin
		set_country_flag = republican_wins_election_1900
		country_event = RON_American_Elections.1
	}
	else_if = {
		limit = { 
			OR = {
			check_variable = { democratic_vote > 270 }
			check_variable = { democratic_vote > republican_vote }				
			}
		}
		set_country_flag = democratic_wins_election_1900
		custom_effect_tooltip = demprojectedwin	
		country_event = RON_American_Elections.1
	}
	else_if = {
		limit = {
			check_variable = { socialist_vote > 270 }
			OR = {
			check_variable = { socialist_vote > republican_vote }
			check_variable = { socialist_vote > democratic_vote }
			}
		}
		set_country_flag = socialist_wins_election_1900
		custom_effect_tooltip = socprojectedwin
		country_event = RON_American_Elections.1
	}
	else = {
		custom_effect_tooltip = noprojectwin
		country_event = RON_American_Elections.1
		add_political_power = -100
	}
	set_country_flag = USA_campaigns_over
}
