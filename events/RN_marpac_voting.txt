country_event = {        #######Resolution 51: Operation Juin
	id = marpac.1
	title = marpac.1.t
	desc = marpac.1.d
	picture = GFX_FRA_may_pustch
	is_triggered_only = yes
    immediate = {
	set_variable = { global.marpac_for_vote = 0 }
	set_variable = { global.marpac_against_vote = 0 }		
	clear_array = global.marpac_executive_array
		every_country = {
		limit = { 
		OR = {
		has_ideas = superstate_size7
		has_ideas = superstate_size6
		has_ideas = superstate_size5
		AND = {
			NOT = {
				has_idea = foreign_isolationism
				has_idea = isolation
			}
			is_major = yes
		}
		has_ideas = RON_United_Nations_Security_Council_NP
		has_ideas = RON_United_Nations_Security_Council		
		}
		}
		add_to_array = {
    	array = global.marpac_executive_array
    	value = THIS.id
		}
		}	
	}
	option = {
		name = marpac.1.a
		set_global_flag = MARPAC_voting_in_process
		set_global_flag = MARPAC_resolution_51
        set_country_flag = show_marpac_menu			
		hidden_effect = {
		every_country = {
		limit = { 
		OR = {
		has_ideas = superstate_size7
		has_ideas = superstate_size6
		has_ideas = superstate_size5
		AND = {
			NOT = {
				has_ideas = foreign_isolationism
				has_ideas = isolation
			}
			is_major = yes
		}
		has_ideas = RON_United_Nations_Security_Council_NP
		has_ideas = RON_United_Nations_Security_Council		
		}
		}
		if = {
			limit = {
				is_ai = yes
			}
		country_event = { 
		id = marpac.5
		days = 1
		}			
		}
		}	
		
		country_event = {
		id = marpac.2
		days = 15
		}
		}
	}	
}

	id = marpac.5
	title = marpacnews.t
	desc = marpac.5.d
	picture = GFX_FRA_may_pustch
	is_triggered_only = yes
	option = {
		name = marpac.5.a	
		trigger = {
			has_global_flag = MARPAC_resolution_1
			OR = {
			has_war = no
			has_government = neutrality			
			has_war_support > .6	
			owns_state = 729
			owns_state = 326
			owns_state = 1028
			owns_state = 745
			owns_state = 743
			owns_state = 902
			}
			NOT = {
				has_ideas = isolation
				has_ideas = foreign_isolationism
			}
		}
		trigger = {
			has_global_flag = MARPAC_resolution_10
			has_war_support < .5
		}
			set_country_flag = MARPAC_voted	 
		    add_to_array = {
	        array = global.marpac_countries_for
            value = THIS.id
		}		
			add_to_variable = {
	        var = global.marpac_for_vote
	        value = 1
        }	
	}	
	option = {
		name = marpac.5.b			

			set_country_flag = MARPAC_voted	
			add_to_variable = {
	        var = global.marpac_against_vote
	        value = 1
			}
		    add_to_array = {
	        array = global.marpac_countries_against
            value = THIS.id
		}				
	}	
}
